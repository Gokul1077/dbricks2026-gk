pythonfrom pyspark.sql import DataFrame

def clean_column_names(df: DataFrame) -> DataFrame:
    """Strip spaces + lowercase all column names."""
    for col in df.columns:
        df = df.withColumnRenamed(col, col.strip().lower().replace(" ", "_"))
    return df

# ── *args: variable positional arguments ─────────────────
def read_and_union(*paths: str) -> DataFrame:
    """Union multiple DataFrames from ADLS paths."""
    dfs = [spark.read.parquet(p) for p in paths]
    from functools import reduce
    return reduce(lambda a, b: a.union(b), dfs)

# ── **kwargs: variable keyword arguments ─────────────────
def write_delta(df: DataFrame, path: str, **options) -> None:
    """Write DataFrame as Delta with custom options."""
    writer = df.write.format("delta")
    for k, v in options.items():
        writer = writer.option(k, v)
    writer.save(path)

# Usage:

DE Best Practice — Generators vs Lists:

When reading large log files or API responses, use generators (yield) instead of building a full list. A generator doesn't load all data into RAM — it produces items one at a time. Critical for pipelines processing millions of records on a single driver node.

🏦 Banking Example — Parameterized ETL Function
pythondef run_etl_layer(
    spark:       SparkSession,
    source_path: str,
    target_path: str,
    filters:     Dict[str, str] = None,
    write_mode:  str = "overwrite"
) -> Dict[str, int]:
    """
    Generic ETL function for banking pipeline layers.
    Returns row counts before/after filtering.
    """
    # Read
    df = spark.read.format("delta").load(source_path)
    original_count = df.count()

    # Apply optional filters from dict
    if filters:
        for col_name, col_value in filters.items():
            df = df.filter(F.col(col_name) == col_value)

    # Standardize schema
    df = clean_column_names(df)
    filtered_count = df.count()

    # Write
    df.write.format("delta").mode(write_mode).save(target_path)

    return {"original": original_count, "filtered": filtered_count}

# Usage

    write_mode   = "overwrite"
)
print(f"Loaded {stats['filtered']} of {stats['original']} records")

pythonfrom pyspark.sql import SparkSession, DataFrame
from pyspark.sql import functions as F
from typing import Optional, Dict
from abc import ABC, abstractmethod
from datetime import datetime
import logging

# ── BASE CLASS (Abstract) ─────────────────────────────────
class BaseETLPipeline(ABC):
    """Abstract base class for all ETL pipelines."""

    pipeline_version = "2.0"  # class attribute

    def __init__(
        self,
        spark:       SparkSession,
        source_path: str,
        target_path: str,
        pipeline_name: str
    ):
        self.spark         = spark
        self.source_path   = source_path
        self.target_path   = target_path
        self.pipeline_name = pipeline_name
        self.logger        = logging.getLogger(self.__class__.__name__)
        self._run_metadata = {}  # private: _ prefix convention

    # ── Abstract method (MUST be implemented by child class) ─
    @abstractmethod
    def transform(self, df: DataFrame) -> DataFrame:
        pass

    # ── Concrete methods (shared by all child classes) ────────
    def extract(self) -> DataFrame:
        self.logger.info(f"Reading from: {self.source_path}")
        return self.spark.read.format("delta").load(self.source_path)

    def load(self, df: DataFrame, mode: str = "overwrite") -> None:
        df.write.format("delta").mode(mode).save(self.target_path)
        self.logger.info(f"Written to: {self.target_path}")

    def run(self) -> Dict:
        """Orchestrates the full ETL — same for all child classes."""
        start = datetime.now()
        try:
            df_raw    = self.extract()
            df_clean  = self.transform(df_raw)   # polymorphic call
            self.load(df_clean)
            return {"status": "SUCCESS", "rows": df_clean.count()}
        except Exception as e:
            self.logger.error(f"Pipeline failed: {e}")
            return {"status": "FAILED", "error": str(e)}

    # ── Dunder methods ────────────────────────────────────────
    def __repr__(self) -> str:
        return f"{self.__class__.__name__}({self.pipeline_name})"

    def __str__(self) -> str:
        return f"Pipeline: {self.pipeline_name} v{self.pipeline_version}"


# ── CHILD CLASS (Concrete) ────────────────────────────────
class BankingTransactionETL(BaseETLPipeline):
    """Concrete ETL for banking transaction data."""

    def __init__(self, spark, source_path, target_path, filter_status: str = "COMPLETED"):
        super().__init__(spark, source_path, target_path, "BankingTransactionETL")
        self.filter_status = filter_status  # extra attribute

    def transform(self, df: DataFrame) -> DataFrame:
        """Banking-specific transformations."""
        return (
            df
            .filter(F.col("status") == self.filter_status)
            .withColumn("amount_inr", F.col("amount") * F.lit(83.5))
            .withColumn("load_ts", F.current_timestamp())
            .dropDuplicates(["txn_id"])
        )


# ── USAGE ────────────────────────────────────────────────

    spark       = spark,

    target_path = "abfss://curated@datalake.dfs.core.windows.net/transactions",

python# ── Full pipeline tying everything together ───────────────
from pyspark.sql import SparkSession, functions as F
from pyspark.sql.types import StructType, StructField, StringType, DoubleType
from typing import List, Dict

# ── Dict: Spark config ────────────────────────────────────
spark_conf = {
    "spark.sql.shuffle.partitions":      "50",
    "spark.sql.adaptive.enabled":        "true",
    "spark.databricks.delta.optimizeWrite.enabled": "true"
}
builder = SparkSession.builder.appName("BankingETL")
for k, v in spark_conf.items():       # loop over dict
    builder = builder.config(k, v)
spark = builder.getOrCreate()

# ── Tuple: immutable schema definition ───────────────────
TXN_SCHEMA = StructType([
    StructField("txn_id",      StringType(), False),
    StructField("amount",      DoubleType(), True),
    StructField("currency",    StringType(), True),
    StructField("status",      StringType(), True),
])

# ── List: required columns ───────────────────────────────
REQUIRED_COLS: List[str] = ["txn_id", "amount", "currency", "status"]

# ── Function: transformation logic ──────────────────────
def enrich_transactions(df) -> object:
    rename_map: Dict[str, str] = {      # dict: column mapping
        "txn_id":   "transaction_id",
        "currency": "currency_code"
    }
    valid_statuses = {"COMPLETED", "PENDING"}  # set: membership check

    # Schema validation using set difference
    missing = set(REQUIRED_COLS) - set(df.columns)
    if missing:
        raise ValueError(f"Missing: {missing}")

    # Filter using set membership (converted to list for Spark)
    df = df.filter(F.col("status").isin(list(valid_statuses)))

    # Rename using dict comprehension → loop
    for old, new in rename_map.items():
        df = df.withColumnRenamed(old, new)

    # Select using list
    output_cols = ["transaction_id", "amount", "currency_code", "status"]
    return df.select(output_cols)

# ── Read → Transform → Write ─────────────────────────────
df_raw  = spark.read.schema(TXN_SCHEMA).parquet("abfss://raw@lake.dfs.core.windows.net/txn/")
df_out  = enrich_transactions(df_raw)
df_out.write.format("delta").mode("overwrite").save("abfss://curated@lake.dfs.core.windows.net/txn/")

🎤 Interview Insights & Quiz

Most-asked Python DS questions in Data Engineering interviews — phrased 3 ways with model answers.

🎤 Interview Q&A — Python Data Structures for DE

Q1: "What's the difference between a list and a tuple, and when would you use each in a pipeline?"
Use a list when data can change — like a list of files to process that you add to dynamically. Use a tuple for fixed things — like database connection parameters, schema definitions, or function return values. Tuples are also faster and can be used as dictionary keys. In production, I use tuples for Spark schema field definitions so they can't be accidentally mutated mid-pipeline.
Q2: "How would you validate schema using Python data structures before writing to Delta Lake?"
I'd use a set difference: required = {'txn_id','amount','status'}, missing = required - set(df.columns). If missing is non-empty, I raise a SchemaValidationError before the write. This is O(1) lookup vs O(n) for list membership, which matters for wide schemas.
Q3: "Why would you wrap a Spark ETL in a class rather than writing flat functions?"
Classes let you: (1) encapsulate config — source/target paths, credentials — so they don't leak into global scope; (2) use inheritance — a base ETL class handles extract/load while child classes override only the transform logic; (3) support testing — you can mock the class and test transform() in isolation. In production, each table's pipeline is a separate class inheriting from a base, making it easy to onboard new tables.
Q4 (Trick): "Can you mutate a tuple in Python?"
The tuple itself can't be reassigned, but if it contains a mutable object like a list, that inner object can be modified. Example: t = ([1,2], [3,4]) → t[0].append(99) works. In DE, this means tuple-stored configs can still be vulnerable if they contain mutable containers — always use immutable types (strings, ints) inside tuples for true safety.
Q5: "When would you use a dictionary vs a DataFrame for config in PySpark?"
Dict for pipeline config / options that the driver needs (Spark read options, column rename maps, connection strings). These are small, driver-side structures. DataFrame for actual data that needs distributed processing. A common mistake is loading lookup tables into Python dicts for use in UDFs — for large lookups, a Spark broadcast join is far more efficient.
🚨 Common Mistakes
  • Using {} to create an empty set → it creates an empty dict. Use set().
  • Forgetting the trailing comma for single-element tuples: ("only") is a string, not a tuple. Use ("only",).
  • Using a list for O(n) membership checks inside UDFs — use a set or dict for O(1) lookup.
  • Mutating a list while iterating over it — always iterate a copy: for item in list(my_list):
  • Not using *args / **kwargs for flexible pipeline functions — leads to 10-parameter function signatures.
🧠 Quiz Question
A Databricks pipeline reads 500,000 banking transactions daily. You need to check if each transaction's currency_code is in a list of 150 approved currencies. What Python data structure should you use for the lookup, and why?
  • A) A Python list — because it preserves insertion order and is easy to iterate
  • B) A Python set — because membership lookup is O(1) via hash table vs O(n) for list
  • C) A Python tuple — because it's immutable and slightly faster than a list
  • D) A Spark DataFrame — because all data should stay in distributed storage
✅ Correct Answer: B) Python set

A set uses a hash table internally, making membership checks O(1) — constant time regardless of size. Checking 500,000 records against a list of 150 items would be O(n × m) = 75 million comparisons. With a set, it's O(n × 1) = 500,000 lookups. For even better Spark-native performance at scale, convert it to a broadcast variable and use df.filter(F.col("currency").isin(list(approved_set))) — Spark broadcasts small lookup collections to all executors.

💡 Pro Tips
  • Use dataclasses or Pydantic models instead of raw dicts for pipeline configs — you get type validation, IDE autocomplete, and clear error messages when a required field is missing. Critical for production pipelines maintained by teams.
  • Never use Python UDFs for heavy computation — they serialize each row across the Py4J bridge. Use Spark built-in functions (pyspark.sql.functions) or Pandas UDFs (vectorized) whenever possible. Python UDFs are 10–100× slower.
  • Profile your partition count after wide transformations. Your Spark notes mention spark.sql.shuffle.partitions=200 as the default. For small daily banking batches (< 1GB), reduce this to 20–50 to avoid 200 tiny output files that kill downstream query performance.
